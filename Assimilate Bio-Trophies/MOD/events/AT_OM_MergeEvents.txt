namespace = event_RD_RR_OM_merge

country_event = {
	id = event_RD_RR_OM_merge.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		any_owned_pop_species = {
			count_species_pop = { count > 0 }
			has_living_standard = {
				type = LivingStandard_RD_RR_OM_TrophyMerge
				country = root
			}
		}
	}
	immediate = {
		ordered_owned_pop_species = {
			limit = {
				NOT = { is_same_species = root.owner_species }
				home_planet = { is_planet = root.owner_species.home_planet }
				is_robotic = no
			}
			order_by = trigger:num_trait_points
			position = 0
			save_event_target_as = EventTarget_RD_RR_OM_MainTrophySpecies
		}
		every_owned_planet = {
			limit = {
				any_owned_species = {
					has_living_standard = {
						type = LivingStandard_RD_RR_OM_TrophyMerge
						country = root
					}
				}
			}
			random_list = {
				1 = { # 3 pops
					set_variable = {
						which = variable_RD_RR_OM_MergeMaximum
						value = 3
					}
				}
				1 = { # 6 pops
					set_variable = {
						which = variable_RD_RR_OM_MergeMaximum
						value = 6
					}
				}
				1 = { # 12 pops
					set_variable = {
						which = variable_RD_RR_OM_MergeMaximum
						value = 12
					}
				}
			}
			while = {
				count = variable_RD_RR_OM_MergeMaximum
				limit = {
					check_variable = {
						which = variable_RD_RR_OM_MergeMaximum
						value > 0
					}
					any_owned_pop = {
						has_living_standard = {
							type = LivingStandard_RD_RR_OM_TrophyMerge
							country = root
						}
					}
				}
				# similar to assimilation, this prefers to process as many of the same species as possible before re-randomizing
				random_owned_pop = {
					limit = {
						has_living_standard = {
							type = LivingStandard_RD_RR_OM_TrophyMerge
							country = root
						}
					}
					species = { save_event_target_as = EventTarget_RD_RR_OM_MergingSpecies }
					change_species = event_target:EventTarget_RD_RR_OM_MainTrophySpecies
					set_living_standard = {
						type = living_standard_organic_trophy
						country = owner
					}
					planet = {
						change_variable = {
							which = variable_RD_RR_OM_MergeMaximum
							value = -1
						}
						while = {
							limit = {
								check_variable = {
									which = variable_RD_RR_OM_MergeMaximum
									value > 0
								}
								any_owned_pop = { is_exact_same_species = event_target:EventTarget_RD_RR_OM_MergingSpecies }
							}
							random_owned_pop = {
								limit = { is_exact_same_species = event_target:EventTarget_RD_RR_OM_MergingSpecies }
								change_species = event_target:EventTarget_RD_RR_OM_MainTrophySpecies
								set_living_standard = {
									type = living_standard_organic_trophy
									country = owner
								}
								planet = {
									change_variable = {
										which = variable_RD_RR_OM_MergeMaximum
										value = -1
									}
								}
							}
						}
					}
				}
			}
			clear_variable = variable_RD_RR_OM_MergeMaximum
		}
	}
}
